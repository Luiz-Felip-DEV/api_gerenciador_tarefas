  `name` varchar(100) DEFAULT NULL,

  `email` varchar(100) DEFAULT NULL,
  `password` varchar(100) DEFAULT NULL,
  `date_register` date DEFAULT curdate(),

  CONSTRAINT `check_senha_minimo_caracteres` CHECK (octet_length(`password`) >= 8),
